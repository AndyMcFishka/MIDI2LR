MIDI2LR  2.0.0.0
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6/CC. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

77 void LR_IPC_OUT::handleMIDI(RSJ::Message mm) {
78  MessageType mt;
79  switch (mm.MessageType) {//this is needed because mapping uses custom structure
80  case RSJ::kCCFlag:
81  mt = CC;
82  break;
83  case RSJ::kNoteOnFlag:
84  mt = NOTE;
85  break;
86  case RSJ::kPWFlag:
87  mt = PITCHBEND;
88  break;
89  default:
90  assert(0);//shouldn't have non-handled message type
91  mt = CC;
92  }
93 //used to handling channel numbers 1-based
94  MIDI_Message_ID message{mm.Channel + 1, mm.Number, mt};
95 
96  if (command_map_) {
97  if (!command_map_->messageExistsInMap(message) ||
98  command_map_->getCommandforMessage(message) == "Unmapped" ||
99  find(LRCommandList::NextPrevProfile.begin(),
100  LRCommandList::NextPrevProfile.end(),
101  command_map_->getCommandforMessage(message)) != LRCommandList::NextPrevProfile.end())
102  return;
103 
104  auto command_to_send = command_map_->getCommandforMessage(message);
105  double computed_value = controls_model_->ControllerToPlugin(mm.MessageType, mm.Channel,
106  mm.Number, mm.Value);
107  command_to_send += ' ' + std::to_string(computed_value) + '\n';
108  {
109  std::lock_guard<decltype(command_mutex_)> lock(command_mutex_);
110  command_ += command_to_send;
111  }
112  juce::AsyncUpdater::triggerAsyncUpdate();
113  }
114 }
