MIDI2LR  2.2.0.0
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6/CC. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

159 int ms::SurfaceStack::frames_pending(mc::CompositorID id) const
160 {
161  RecursiveReadLock lg(guard);
162 
163  int result = scene_changed ? 1 : 0;
164  for (auto const& surface : surfaces)
165  {
166  if (surface->visible())
167  {
168  auto const tracker = rendering_trackers.find(surface.get());
169  if (tracker != rendering_trackers.end() && tracker->second->is_exposed_in(id))
170  {
171  // Note that we ask the surface and not a Renderable.
172  // This is because we don't want to waste time and resources
173  // on a snapshot till we're sure we need it...
174  int ready = surface->buffers_ready_for_compositor(id);
175  if (ready > result)
176  result = ready;
177  }
178  }
179  }
180  return result;
181 }

323 {

329  RecursiveWriteLock ul(guard);

331
