MIDI2LR  2.0.0.0
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6/CC. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

71 void CommandMenu::clicked(const juce::ModifierKeys& modifiers) {
72  if (modifiers.isPopupMenu()) {
73  switch (message_.messageType) {
74  case CC:
75  {
76  CCoptions ccopt;
77  ccopt.bindToControl(static_cast<size_t>(message_.channel) - 1, // convert 1-based to 0-based //-V201
78  static_cast<short>(message_.controller)); //-V201
79  juce::DialogWindow::showModalDialog("Adjust CC dialog", &ccopt, nullptr,
80  juce::Colour::fromRGB(0xFF, 0xFF, 0xFF), true);
81  break;
82  }
83  case PITCHBEND:
84  {
85  PWoptions pwopt;
86  pwopt.bindToControl(static_cast<size_t>(message_.channel) - 1); //-V201 convert 1-based to 0 based
87  juce::DialogWindow::showModalDialog("Adjust PW dialog", &pwopt, nullptr,
88  juce::Colour::fromRGB(0xFF, 0xFF, 0xFF), true);
89  break;
90  }
91  default:
92  /* do nothing for other types of controllers */;
93  }
94  }
95  else {
96  size_t index = 1;
97  auto submenu_tick_set = false;
98  juce::PopupMenu main_menu;
99  main_menu.addItem(index, "Unmapped", true, submenu_tick_set = (index == selected_item_));
100  index++;
101 // add each submenu
102  for (size_t menu_index = 0; menu_index < menus_.size(); ++menu_index) {
103  juce::PopupMenu subMenu;
104  for (const auto& command : menu_entries_[menu_index]) {
105  auto already_mapped = false;
106  if ((index - 1 < LRCommandList::LRStringList.size()) && (command_map_)) {
107  already_mapped =
108  command_map_->commandHasAssociatedMessage(LRCommandList::LRStringList[index - 1]);
109  }
110 
111  // add each submenu entry, ticking the previously selected entry and
112  // disabling a previously mapped entry
113 
114  if (already_mapped)
115  subMenu.addColouredItem(static_cast<int>(index), command, juce::Colours::red, true, //-V202
116  index == selected_item_);
117  else
118  subMenu.addItem(static_cast<int>(index), command, true, index == selected_item_); //-V202
119 
120  index++;
121  }
122  // set whether or not the submenu is ticked (true if one of the submenu's
123  // entries is selected)
124  main_menu.addSubMenu(menus_[menu_index], subMenu, true, nullptr,
125  selected_item_ < index && !submenu_tick_set);
126  submenu_tick_set |= (selected_item_ < index && !submenu_tick_set);
127  }
128 
129  const auto result = static_cast<size_t>(main_menu.show());
130  if ((result) && (command_map_)) {
131  // user chose a different command, remove previous command mapping
132  // associated to this menu
133  if (selected_item_ < std::numeric_limits<size_t>::max())
134  command_map_->removeMessage(message_);
135 
136  if (result - 1 < LRCommandList::LRStringList.size())
137  setButtonText(LRCommandList::LRStringList[result - 1]);
138  else
139  setButtonText(LRCommandList::NextPrevProfile[result - 1 - LRCommandList::LRStringList.size()]);
140 
141  selected_item_ = result;
142 
143  // Map the selected command to the CC
144  command_map_->addCommandforMessage(result - 1, message_);
145  }
146  }
147 }
