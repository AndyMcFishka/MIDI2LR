MIDI2LR  2.0.4.0
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6/CC. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

33 void MIDISender::sendCC(int midi_channel, int controller, int value) const {
34  if (controller < 128) { // regular message
35  for (const auto& dev : output_devices_) {
36  dev->sendMessageNow(juce::MidiMessage::controllerEvent(midi_channel, controller, value));
37  }
38  }
39  else { // NRPN
40  const auto parameterLSB = controller & 0x7f;
41  const auto parameterMSB = (controller >> 7) & 0x7F;
42  const auto valueLSB = value & 0x7f;
43  const auto valueMSB = (value >> 7) & 0x7F;
44  for (const auto& dev : output_devices_) {
45  dev->sendMessageNow(juce::MidiMessage::controllerEvent(midi_channel, 99, parameterMSB));
46  dev->sendMessageNow(juce::MidiMessage::controllerEvent(midi_channel, 98, parameterLSB));
47  dev->sendMessageNow(juce::MidiMessage::controllerEvent(midi_channel, 6, valueMSB));
48  dev->sendMessageNow(juce::MidiMessage::controllerEvent(midi_channel, 38, valueLSB));
49  }
50  }
51 }
