MIDI2LR  5.4.0.0
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6+/CC Classic. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

63  constexpr MessageType ToMessageType(std::underlying_type_t<MessageType> value)
64  {
65  static_assert(std::is_unsigned_v<decltype(value)>, "Avoid sign extension");
66  const auto from {value >> 4U & 0xFU};
67  if (std::cmp_less(
68  from, static_cast<std::underlying_type_t<MessageType>>(MessageType::kNoteOff))) {
69  throw std::out_of_range("ToMessageType: MessageType range error, must be 0x8 to 0xF");
70  }
71  return static_cast<MessageType>(from);
72  }

95 template<typename Char> struct fmt::formatter<rsj::MessageType, Char> {
96  template<typename ParseContext> constexpr auto parse(ParseContext& ctx)
97  { /* parsing copied from fmt's chrono.h */
98  auto it {ctx.begin()};
99  if (!it) { return ctx.end(); }
100  if (it != ctx.end() && *it == ':') { std::advance(it, 1); }
101  auto end {std::find(it, ctx.end(), '}')};
102  tm_format_.reserve(detail::to_unsigned(end - it + 1));
103  tm_format_.append(it, end);
104  tm_format_.push_back('\0');
105  return end;
106  }
107 
108  template<typename FormatContext> auto format(const rsj::MessageType& p, FormatContext& ctx)
109  {
110  if (tm_format_[0] == 'n') {
111  return fmt::format_to(ctx.out(), "{}", rsj::MessageTypeToName(p));
112  }
113  return fmt::format_to(ctx.out(), "{}", rsj::MessageTypeToLabel(p));
114  }

117  basic_memory_buffer<Char> tm_format_;
