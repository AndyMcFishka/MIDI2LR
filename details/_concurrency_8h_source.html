MIDI2LR  4.0.4.0
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6+/CC Classic. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

37  class PRNG {
38  public:
39  using result_type = uint64_t;
40  static result_type NextRandom() noexcept
41  {
42  result_type z {
43  state_.fetch_add(0x9e3779b97f4a7c15, std::memory_order_relaxed) + 0x9e3779b97f4a7c15};
44  z = (z ^ z >> 30) * 0xbf58476d1ce4e5b9;
45  z = (z ^ z >> 27) * 0x94d049bb133111eb;
46  return z ^ z >> 31;
47  static_assert(std::is_unsigned_v<decltype(z)>, "Avoid sign extension");
48  }

50  {

52  }
53  [[nodiscard]] constexpr result_type max() const noexcept
54  {
55  return std::numeric_limits<result_type>::max();
56  }
57  result_type operator()() noexcept
58  {
59  return NextRandom();
60  }

75  class SpinLock {
76  public:
77  SpinLock() noexcept = default;
78  ~SpinLock() = default;
79  SpinLock(const SpinLock& other) = delete;
80  SpinLock(SpinLock&& other) = delete;
81  SpinLock& operator=(const SpinLock& other) = delete;
82  SpinLock& operator=(SpinLock&& other) = delete;
83  void lock() noexcept
84  { /* if uncontested, don't bother with exponential back-off */
85  if (!flag_.load(std::memory_order_relaxed)
86  && !flag_.exchange(true, std::memory_order_acquire))
87  return;
88  using LoopT = std::uint64_t;
89  const auto [bo1, bo2, bo3] {[]() noexcept -> std::tuple<LoopT, LoopT, LoopT> {
90  static_assert(std::is_unsigned_v<
91  PRNG::result_type> && sizeof(PRNG::result_type) >= sizeof(uint16_t));
92  const auto rn {PRNG::NextRandom()};
93  return {1 + (rn & 0b11), 12 + (rn >> 2 & 0b111), 56 + (rn >> 5 & 0b1111)};
94  }()};
95  do {
96  /* avoid cache invalidation if lock appears to be unavailable */
97  for (LoopT k {0}; flag_.load(std::memory_order_relaxed); ++k) {
98  using namespace std::chrono_literals;
99  if (k < bo1)
100  continue;
101  if (k < bo2)
102  _mm_pause();
103  else if (k < bo3)
104  std::this_thread::yield();
105  else
106  std::this_thread::sleep_for(1ms);
107  }
108  } while (flag_.exchange(true, std::memory_order_acquire));
109  }
110  bool try_lock() noexcept
111  {
112  /* avoid cache invalidation if lock appears to be unavailable */
113  if (flag_.load(std::memory_order_relaxed))
114  return false;
115  return !flag_.exchange(true, std::memory_order_acquire); /* try to acquire lock */
116  }
117  void unlock() noexcept
118  {
119  flag_.store(false, std::memory_order_release);
120  }
121 
122  private: /* see https://stackoverflow.com/a/52158819/5699329 for 128 value */
123  alignas(128) std::atomic<bool> flag_ {false};
124  static_assert(sizeof(bool) == 1, "Padding assumes size of bool is 1.");
125  // NOLINTNEXTLINE(cppcoreguidelines-avoid-c-arrays, modernize-avoid-c-arrays)
126  [[maybe_unused]] alignas(1) std::byte padding_[127];
127  };
