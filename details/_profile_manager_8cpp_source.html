MIDI2LR  2.0.4.0
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6/CC. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

125 void ProfileManager::MIDIcmdCallback(RSJ::Message mm) {
126  MessageType mt;
127  switch (mm.MessageType) {//this is needed because mapping uses custom structure
128  case RSJ::kCCFlag:
129  mt = CC;
130  break;
131  case RSJ::kNoteOnFlag:
132  mt = NOTE;
133  break;
134  case RSJ::kPWFlag:
135  mt = PITCHBEND;
136  break;
137  default: //should be unreachable
138  assert(0);
139  mt = CC;
140  }
141  //used to handling Channel as 1-based
142  const MIDI_Message_ID cc{ mm.Channel + 1, mm.Number, mt };
143 
144  if (command_map_) {
145  // return if the value isn't high, or the command isn't a valid
146  // profile-related command
147  if ((controls_model_->ControllerToPlugin(mm.MessageType, mm.Channel, mm.Number, mm.Value) < 0.99)
148  || !command_map_->messageExistsInMap(cc))
149  return;
150 
151  mapCommand(cc);
152  }
153 }
