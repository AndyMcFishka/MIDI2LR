MIDI2LR  2.4.0.0
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6/CC. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

191 void CommandTableModel::buildFromXml(const juce::XmlElement * const root)
192 {
193  if (root->getTagName().compare("settings") != 0)
194  return;
195  removeAllRows();
196  const auto* setting = root->getFirstChildElement();
197  while ((setting) && (command_map_)) {
198  if (setting->hasAttribute("controller")) {
199  const RSJ::MidiMessageId message{setting->getIntAttribute("channel"),
200  setting->getIntAttribute("controller"), RSJ::MsgIdEnum::CC};
201  addRow(message.channel, message.controller, RSJ::MsgIdEnum::CC);
202  command_map_->addCommandforMessage(setting->
203  getStringAttribute("command_string").toStdString(), message);
204  }
205  else if (setting->hasAttribute("note")) {
206  const RSJ::MidiMessageId note{setting->getIntAttribute("channel"),
207  setting->getIntAttribute("note"), RSJ::MsgIdEnum::NOTE};
208  addRow(note.channel, note.pitch, RSJ::MsgIdEnum::NOTE);
209  command_map_->addCommandforMessage(setting->
210  getStringAttribute("command_string").toStdString(), note);
211  }
212  else if (setting->hasAttribute("pitchbend")) {
213  const RSJ::MidiMessageId pb{setting->getIntAttribute("channel"), 0,
214  RSJ::MsgIdEnum::PITCHBEND};
215  addRow(pb.channel, 0, RSJ::MsgIdEnum::PITCHBEND);
216  command_map_->addCommandforMessage(setting->
217  getStringAttribute("command_string").toStdString(), pb);
218  }
219  setting = setting->getNextElement();
220  }
221  Sort();
222 }
