MIDI2LR  2.3.0.0
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6/CC. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

26 double ChannelModel::ControllerToPlugin(short controltype, size_t controlnumber, short value) noexcept(ndebug)
27 {
28  Expects((controltype == RSJ::kCCFlag && ccMethod_[controlnumber] == RSJ::CCmethod::absolute) ? (ccLow_[controlnumber] < ccHigh_[controlnumber]) : 1);
29  Expects((controltype == RSJ::kPWFlag) ? (pitchWheelMax_ > pitchWheelMin_) : 1);
30  Expects((controltype == RSJ::kPWFlag) ? value >= pitchWheelMin_ && value <= pitchWheelMax_ : 1);
31  //note that the value is not msb,lsb, but rather the calculated value. Since lsb is only 7 bits, high bits are shifted one right when placed into short.
32  switch (controltype) {
33  case RSJ::kPWFlag:
34  return static_cast<double>(value - pitchWheelMin_) / static_cast<double>(pitchWheelMax_ - pitchWheelMin_);
35  case RSJ::kCCFlag:
36  switch (ccMethod_[controlnumber]) {
37  case RSJ::CCmethod::absolute:
38  return static_cast<double>(value - ccLow_[controlnumber]) / static_cast<double>(ccHigh_[controlnumber] - ccLow_[controlnumber]);
39  case RSJ::CCmethod::binaryoffset:
40  if (IsNRPN_(controlnumber))
41  return OffsetResult_(value - kBit14, controlnumber);
42  return OffsetResult_(value - kBit7, controlnumber);
43  case RSJ::CCmethod::signmagnitude:
44  if (IsNRPN_(controlnumber))
45  return OffsetResult_((value & kBit14) ? -(value & kLow13Bits) : value, controlnumber);
46  return OffsetResult_((value & kBit7) ? -(value & kLow6Bits) : value, controlnumber);
47  case RSJ::CCmethod::twoscomplement: //see https://en.wikipedia.org/wiki/Signed_number_representations#Two.27s_complement
48  if (IsNRPN_(controlnumber)) //flip twos comp and subtract--independent of processor architecture
49  return OffsetResult_((value & kBit14) ? -((value ^ kMaxNRPN) + 1) : value, controlnumber);
50  return OffsetResult_((value & kBit7) ? -((value ^ kMaxMIDI) + 1) : value, controlnumber);
51  default:
52  Expects(!"Should be unreachable code in ControllerToPlugin--unknown CCmethod");
53  return 0.0;
54  }
55  case RSJ::kNoteOnFlag:
56  return static_cast<double>(value) / static_cast<double>((IsNRPN_(controlnumber) ? kMaxNRPN : kMaxMIDI));
57  case RSJ::kNoteOffFlag:
58  return 0.0;
59  default:
60  Expects(!"Should be unreachable code in ControllerToPlugin--unknown control type");
61  return 0.0;
62  }
63 }

65 short ChannelModel::PluginToController(short controltype, size_t controlnumber, double pluginV) noexcept(ndebug)
66 {
67  Expects(controlnumber <= kMaxNRPN);
68  Expects(pluginV >= 0.0 && pluginV <= 1.0);
69  switch (controltype) {
70  case RSJ::kPWFlag:
71  return static_cast<short>(round(pluginV * (pitchWheelMax_ - pitchWheelMin_))) + pitchWheelMin_;
72  case RSJ::kCCFlag:
73  {
74  if (ccMethod_[controlnumber] == RSJ::CCmethod::absolute)
75  return static_cast<short>(round(pluginV *
76  (ccHigh_[controlnumber] - ccLow_[controlnumber]))) + ccLow_[controlnumber];
77  const auto cv = static_cast<short>(round(pluginV * ccHigh_[controlnumber])); //ccLow == 0 for non-absolute
78  if (RSJ::now_ms() - kUpdateDelay > lastUpdate_.load(std::memory_order_acquire))
79  currentV_[controlnumber].store(cv, std::memory_order_release);
80  return cv;
81  }
82  case RSJ::kNoteOnFlag:
83  return kMaxMIDI;
84  default:
85  Expects(!"Unexpected control type");
86  }
87  return 0;
88 }
