MIDI2LR  2.0.4.0
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6/CC. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

218 inline double ChannelModel::OffsetResult_(short diff, size_t controlnumber) noexcept(ndebug) {
219  assert(ccHigh_[controlnumber] > 0); //CCLow will always be 0 for offset controls
220  assert(diff <= kMaxNRPN && diff >= -kMaxNRPN);
221  assert(controlnumber <= kMaxNRPN);
222  lastUpdate_.store(RSJ::now_ms(), std::memory_order_release);
223  short cv = currentV_[controlnumber].fetch_add(diff, std::memory_order_relaxed) + diff;
224  if (cv < 0) {//fix currentV unless another thread has already altered it
225  currentV_[controlnumber].compare_exchange_strong(cv, static_cast<short>(0),
226  std::memory_order_relaxed, std::memory_order_relaxed);
227  return 0.0;
228  }
229  if (cv > ccHigh_[controlnumber]) {//fix currentV unless another thread has already altered it
230  currentV_[controlnumber].compare_exchange_strong(cv, ccHigh_[controlnumber],
231  std::memory_order_relaxed, std::memory_order_relaxed);
232  return 1.0;
233  }
234  return static_cast<double>(cv) / static_cast<double>(ccHigh_[controlnumber]);
235 }
