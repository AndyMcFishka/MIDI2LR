MIDI2LR  2.0.0.0
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6/CC. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

199 void MainContentComponent::handleMIDI(RSJ::Message mm) {
200  // Display the CC parameters and add/highlight row in table corresponding to the CC
201  MessageType mt;
202  switch (mm.MessageType) {//this is needed because mapping uses custom structure
203  case RSJ::kCCFlag:
204  mt = CC;
205  break;
206  case RSJ::kNoteOnFlag:
207  mt = NOTE;
208  break;
209  case RSJ::kPWFlag:
210  mt = PITCHBEND;
211  break;
212  default: //shouldn't receive any messages note categorized above
213  assert(0);
214  mt = CC;
215  }
216  mm.Channel++; //used to 1-based channel numbers
217  last_command_ = juce::String::formatted("%d: CC%d [%d]", mm.Channel, mm.Number, mm.Value);
218  command_table_model_.addRow(mm.Channel, mm.Number, mt);
219  row_to_select_ = static_cast<size_t>(command_table_model_.getRowForMessage(mm.Channel, mm.Number, mt)); //-V201
220  triggerAsyncUpdate();
221 }
