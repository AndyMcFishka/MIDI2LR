MIDI2LR  2.0.0.0
MIDI2LR is an application that interfaces MIDI controllers with Lightroom 6/CC. It processes MIDI input into develop parameter updates and photo actions, and sends MIDI output when parameters are changed for motorized feedback (on controllers that have motorized faders). A listing of available LightRoom commands is in the Wiki. Assistance on the code and design is welcome.

71 void LR_IPC_IN::run() {
72  while (!juce::Thread::threadShouldExit()) {
73  //if not connected, executes a wait 333 then goes back to while
74  //if connected, tries to read a line, checks thread status and connection
75  //status before each read attempt
76  //doesn't terminate thread if disconnected, as currently don't have graceful
77  //way to restart thread
78  if (!juce::StreamingSocket::isConnected()) {
79  juce::Thread::wait(kNotConnectedWait);
80  } //end if (is not connected)
81  else {
82  char line[kBufferSize + 1] = {' '};//plus one for \0 at end
83  auto size_read = 0;
84  auto can_read_line = true;
85  // parse input until we have a line, then process that line, quit if
86  // connection lost
87  while (std::string(line).back() != '\n' && juce::StreamingSocket::isConnected()) {
88  if (juce::Thread::threadShouldExit())
89  goto threadExit;//break out of nested whiles
90  const auto wait_status = juce::StreamingSocket::waitUntilReady(true, kReadyWait);
91  switch (wait_status) {
92  case -1:
93  can_read_line = false;
94  goto dumpLine; //read line failed, break out of switch and while
95  case 0:
96  juce::Thread::wait(kEmptyWait);
97  break; //try again to read until char shows up
98  case 1:
99  if (size_read == kBufferSize)
100  throw std::out_of_range("Buffer overflow in LR_IPC_IN");
101  size_read += juce::StreamingSocket::read(line + size_read, 1, false);
102  break;
103  }
104  } // end while !\n and is connected
105 
106  // if lose connection, line may not be terminated
107  {
108  std::string param{line};
109  if (can_read_line && param.back() == '\n') {
110  processLine(param);
111  }
112  } //scope param
113  dumpLine: /* empty statement */;
114  } //end else (is connected)
115  } //while not threadshouldexit
116 threadExit: /* empty statement */;
117  std::lock_guard< decltype(timer_mutex_) > lock(timer_mutex_);
118  timer_off_ = true;
119  juce::Timer::stopTimer();
120  //thread_started_ = false; //don't change flag while depending upon it
121 }

134 void LR_IPC_IN::processLine(const std::string& line) {
135  const static std::unordered_map<std::string, int> cmds = {
136  {"SwitchProfile",1},
137  {"SendKey",2},
138  {"TerminateApplication",3},
139  };
140  // process input into [parameter] [Value]
141  const auto trimmed_line = RSJ::trim(line);
142  const auto command = trimmed_line.substr(0, trimmed_line.find(' '));
143  const auto value_string = trimmed_line.substr(trimmed_line.find(' ') + 1);
144 
145  switch (cmds.count(command) ? cmds.at(command) : 0) {
146  case 1: //SwitchProfile
147  if (profile_manager_)
148  profile_manager_->switchToProfile(value_string);
149  break;
150  case 2: //SendKey
151  {
152  std::bitset<3> modifiers{static_cast<decltype(modifiers)>
153  (std::stoi(value_string))};
154  send_keys_.SendKeyDownUp(RSJ::ltrim(RSJ::ltrim(value_string, RSJ::digit)),
155  modifiers[0], modifiers[1], modifiers[2]);
156  break;
157  }
158  case 3: //TerminateApplication
159  juce::JUCEApplication::getInstance()->systemRequestedQuit();
160  break;
161  case 0:
162  // send associated messages to MIDI OUT devices
163  if (command_map_ && midi_sender_) {
164  const auto original_value = std::stod(value_string);
165  for (const auto msg : command_map_->getMessagesForCommand(command)) {
166  short msgtype{0};
167  switch (msg->messageType) {
168  case NOTE:
169  msgtype = RSJ::kNoteOnFlag;
170  break;
171  case CC:
172  msgtype = RSJ::kCCFlag;
173  break;
174  case PITCHBEND:
175  msgtype = RSJ::kPWFlag;
176  }
177  const auto value = controls_model_->PluginToController(msgtype,
178  static_cast<size_t>(msg->channel - 1),
179  static_cast<short>(msg->controller), original_value);
180 
181  if (midi_sender_) {
182  switch (msgtype) {
183  case RSJ::kNoteOnFlag: midi_sender_->sendCC(msg->channel, msg->controller, value); break;
184  case RSJ::kCCFlag: midi_sender_->sendCC(msg->channel, msg->controller, value); break;
185  case RSJ::kPWFlag: midi_sender_->sendPitchBend(msg->channel, value); break;
186  }
187  }
188  }
189  }
190  }
